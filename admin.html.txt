        </div>

        <div class="stats-grid" id="stats"></div>

        <div class="filter-bar">
            <button class="filter-btn active" onclick="filterComplaints('all')">All</button>
            <button class="filter-btn" onclick="filterComplaints('Pending')">Pending</button>
            <button class="filter-btn" onclick="filterComplaints('Replied')">Replied</button>
            <button class="filter-btn" onclick="filterComplaints('Resolved')">Resolved</button>
            <button class="filter-btn refresh-btn" onclick="loadComplaints()">🔄 Refresh</button>
        </div>

        <div class="complaints-container" id="complaintsContainer">
            <div class="loading">Loading complaints...</div>
        </div>
    </div>

    <script>
        const API_URL = "/api";
        const ADMIN_USERNAME = "fortinet";
        const ADMIN_PASSWORD = "fortinet123";

        let allComplaints = [];
        let currentFilter = 'all';

        // Check login status
        function checkLogin() {
            if (localStorage.getItem('adminLoggedIn') === 'true') {
                document.getElementById('loginSection').style.display = 'none';
                document.getElementById('dashboardSection').style.display = 'block';
                loadComplaints();
            } else {
                document.getElementById('loginSection').style.display = 'block';
                document.getElementById('dashboardSection').style.display = 'none';
            }
        }

        // Login handler
        document.getElementById('loginForm')?.addEventListener('submit', function(e) {
            e.preventDefault();
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            
            if (username === ADMIN_USERNAME && password === ADMIN_PASSWORD) {
                localStorage.setItem('adminLoggedIn', 'true');
                checkLogin();
            } else {
                document.getElementById('loginError').innerHTML = '<div class="alert alert-error">❌ Invalid credentials</div>';
            }
        });

        async function loadComplaints() {
            const container = document.getElementById('complaintsContainer');
            container.innerHTML = '<div class="loading">Loading complaints...</div>';
            
            try {
                const res = await fetch(`${API_URL}/complaints`);
                const data = await res.json();
                
                if (data.complaints) {
                    allComplaints = data.complaints;
                    updateStats();
                    displayComplaints();
                } else {
                    container.innerHTML = '<div class="loading">No complaints found</div>';
                }
            } catch (err) {
                container.innerHTML = '<div class="loading alert alert-error">Error loading complaints. Make sure server is running.</div>';
            }
        }

        function updateStats() {
            const total = allComplaints.length;
            const pending = allComplaints.filter(c => c.status === 'Pending').length;
            const replied = allComplaints.filter(c => c.status === 'Replied').length;
            const resolved = allComplaints.filter(c => c.status === 'Resolved').length;
            
            document.getElementById('stats').innerHTML = `
                <div class="stat-card"><div class="stat-number">${total}</div><div>Total Complaints</div></div>
                <div class="stat-card"><div class="stat-number">${pending}</div><div>Pending</div></div>
                <div class="stat-card"><div class="stat-number">${replied}</div><div>Replied</div></div>
                <div class="stat-card"><div class="stat-number">${resolved}</div><div>Resolved</div></div>
            `;
        }

        function displayComplaints() {
            let filtered = allComplaints;
            if (currentFilter !== 'all') {
                filtered = allComplaints.filter(c => c.status === currentFilter);
            }
            
            const container = document.getElementById('complaintsContainer');
            if (filtered.length === 0) {
                container.innerHTML = '<div class="loading">No complaints in this category</div>';
                return;
            }
            
            container.innerHTML = '';
            filtered.forEach(c => {
                const isResolved = c.status === 'Resolved';
                container.innerHTML += `
                    <div class="complaint-card ${isResolved ? 'resolved' : ''}">
                        <div class="complaint-header">
                            <span class="complaint-id">🎫 ${c.id}</span>
                            <span class="status-badge status-${c.status}">${c.status}</span>
                        </div>
                        <div class="complaint-details">
                            <p><strong>👤 Customer:</strong> ${c.name} (${c.email})</p>
                            <p><strong>📌 Subject:</strong> ${c.subject}</p>
                            <p><strong>📂 Category:</strong> ${c.category}</p>
                            <p><strong>📝 Complaint Message:</strong><br>${c.message}</p>
                            <p><strong>📅 Date:</strong> ${c.createdAt}</p>
                            ${c.reply ? `<div class="existing-reply"><strong>📝 Your Previous Reply:</strong><br>${c.reply}<br><small>Sent: ${c.replyDate}</small></div>` : ''}
                        </div>
                        <div class="reply-section">
                            <textarea id="reply-${c.id}" class="reply-textarea" placeholder="Type your response to the customer..."></textarea>
                            <div>
                                <button class="btn btn-reply" onclick="sendReply('${c.id}')">📤 Send Reply</button>
                                ${!isResolved ? `<button class="btn btn-resolve" onclick="markResolved('${c.id}')">✅ Mark as Resolved</button>` : ''}
                                <button class="btn btn-delete" onclick="deleteComplaint('${c.id}')">🗑️ Delete Complaint</button>
                            </div>
                        </div>
                    </div>
                `;
            });
        }

        async function sendReply(complaintId) {
            const reply = document.getElementById(`reply-${complaintId}`).value.trim();
            if (!reply) return alert('Please enter a reply message');
            
            const btn = event.target;
            btn.disabled = true;
            btn.textContent = 'Sending...';
            
            try {
                await fetch(`${API_URL}/complaints/${complaintId}/reply`, {
                    method: 'PUT',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ reply: reply })
                });
                alert('✅ Reply sent successfully! Customer can now see it when tracking.');
                loadComplaints();
            } catch (err) {
                alert('❌ Error sending reply');
                btn.disabled = false;
                btn.textContent = '📤 Send Reply';
            }
        }

        async function markResolved(complaintId) {
            if (!confirm('Mark this complaint as resolved?')) return;
            
            const btn = event.target;
            btn.disabled = true;
            btn.textContent = 'Updating...';
            
            try {
                await fetch(`${API_URL}/complaints/${complaintId}/resolve`, {
                    method: 'PUT'
                });
                alert('✅ Complaint marked as resolved!');
                loadComplaints();
            } catch (err) {
                alert('❌ Error updating status');
                btn.disabled = false;
                btn.textContent = '✅ Mark as Resolved';
            }
        }

        async function deleteComplaint(complaintId) {
            if (!confirm('⚠️ Delete this complaint permanently? This cannot be undone.')) return;
            
            try {
                await fetch(`${API_URL}/complaints/${complaintId}`, {
                    method: 'DELETE'
                });
                alert('✅ Complaint deleted successfully!');
                loadComplaints();
            } catch (err) {
                alert('❌ Error deleting complaint');
            }
        }

        function filterComplaints(filter) {
            currentFilter = filter;
            document.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');
            displayComplaints();
        }

        function logout() {
            localStorage.removeItem('adminLoggedIn');
            checkLogin();
        }

        // Initialize
        checkLogin();
        setInterval(() => {
            if (localStorage.getItem('adminLoggedIn') === 'true') {
                loadComplaints();
            }
        }, 30000);
    </script>
